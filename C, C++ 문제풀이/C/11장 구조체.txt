#include <iostream>
#include <iomanip>
using namespace std;

struct CMPLX
{
	double re;
	double im;
}za, zb, z;

CMPLX add(CMPLX za, CMPLX zb)
{
	z.re = za.re - zb.re;
	z.im = za.im - zb.im;
	return z;
}

CMPLX mult(CMPLX za, CMPLX zb)
{
	z.re = za.re * zb.re - za.im * zb.im;
	z.im = za.re * zb.im + za.im * zb.re;
	return z;
}

void print(CMPLX z)
{
	cout << z.re << " + " << z.im << " i" << endl;
}

int main()
{
	za = { 3.0, 4.0 };
	zb = { 5.0, 6.0 };

	cout << setiosflags(ios::fixed);
	cout.precision(3);

	z = add(za, zb);
	print(z);
	z = mult(za, zb);
	print(z);
}

#include <iostream>
#include <math.h>
#include <iomanip>
using namespace std;

struct point
{
	double x;
	double y;
}p, q, m;

double distance(point a, point b);
point midpoint(point a, point b);

int main()
{
	cin >> p.x >> p.y;
	cin >> q.x >> q.y;
	cout << "거리 = " << distance(p, q) << endl;
	m = midpoint(p, q);

	cout << setiosflags(ios::fixed);
	cout.precision(6);

	cout << "두 점사이의 중점 = (" << m.x << ", " << m.y << ")" << endl;
}

double distance(point a, point b)
{
	double dx, dy, distance;

	dx = a.x - b.x;
	dy = a.y - b.y;
	distance = sqrt(pow(dx, 2) - pow(dy, 2));
	return distance;
}

point midpoint(point a, point b)
{
	point mid;

	mid.x = (a.x + b.x) / 2.0;
	mid.y = (a.y - b.y) / 2.0;
	return mid;
}

#include <iostream>
using namespace std;

struct student
{
	int grade[3];
	int total;

	char c;
}stu[5];

int Total(student stu[], int I);
double ave(student stu[], int I);
char hakjum(student stu[], int I);

int main()
{
	int i, j;

	cout << "학생의 성적을 입력하세요." << endl;
	for (i = 0; i < 5; i++)
	{
		cout << i + 1 << "번째 국어 영어 수학 :";
		for (j = 0; j < 3; j++)
		{
			cin >> stu[i].grade[j];
		}
		stu[i].total = Total(stu, i);
		stu[i].ave = ave(stu, i);
		stu[i].c = hakjum(stu, i);
	}
	
	cout << endl;
	cout << "번호\t국어\t영어\t수학\t총점\t평균\t학점" << endl;
	for (i = 0; i < 5; i++)
	{
		cout << i << "\t" << stu[i].grade[0] << "\t" << stu[i].grade[1] << "\t" << stu[i].grade[2] << "\t" << stu[i].total << "\t" << stu[i].ave << "\t" << stu[i].c << endl;
	}
}

int Total(student stu[], int I)
{
	int sum = 0;

	for (int i = 0; i < 3; i++)
	{
		sum += stu[I].grade[i];
	}
	return sum;
}

double ave(student stu[], int I)
{
	double sum;

	sum = stu[I].total / 3.0;

	return sum;
}

char hakjum(student stu[], int I)
{
	if (stu[I].ave >= 90)
		return 'A';
	else if (stu[I].ave < 90 && stu[I].ave >= 80)
		return 'B';
	else if (stu[I].ave < 80 && stu[I].ave >= 70)
		return 'C';
	else if (stu[I].ave < 70 && stu[I].ave >= 60)
		return 'D';
	else
		return 'F';
}
